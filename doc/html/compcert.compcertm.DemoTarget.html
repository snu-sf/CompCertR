Definition lb0: label := 1%positive.
Definition code: list instruction :=

    Ptestq_rr RDI RDI ;

      Pcvtsl2sd_fr XMM0 RDI ;
      Pret ;

      Pmov_rr RAX RDI ;

      Pxorpd_f XMM0 ;
      Pcvtsl2sd_fr XMM0 RAX ;
      Paddd_ff XMM0 XMM0 ;

  ].
