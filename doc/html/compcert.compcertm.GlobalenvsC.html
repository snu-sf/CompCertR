Theorem find_def_match_backward:
  forall b tg, Genv.find_def (Genv.globalenv tp) b = Some tg ->
  exists g, Genv.find_def (Genv.globalenv p) b = Some g /\ match_globdef match_fundef match_varinfo ctx g tg.
