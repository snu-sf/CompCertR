Variable beta : radix.

Notation bpow := (bpow beta).

Variable emin : Z.

Definition FIX_format (x : R) :=
  exists f : float beta,
  x = F2R f /\ (Fexp f = emin)%Z.

Definition FIX_exp (e : Z) := emin.
